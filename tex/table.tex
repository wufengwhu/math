\noindent\textbf{Algorithm 1} \quad \texttt{ALS for $Matrix$ Completion}\\
\noindent\rule{\textwidth}{0.2mm}
\indent Initialize $X$, $Y$ \\ 
\indent \textbf{repeat}\\
\indent\indent \textbf{for $u=1 \ldots n$ do} \\
% \vspace{0.1mm}
\begin{equation} \label{c3}
    x_{u} = (\sum_{r_{ui} \in r_{u*}}^{} y_{i}y_{i}^{T}
        + \lambda I_{k})^{-1}\sum_{r_{ui} \in r_{u*}}r_{ui}y_{i}
\end{equation}
\indent\indent \textbf{end for}\\
\indent\indent \textbf{for $i=1 \ldots m$ do} \\
\begin{equation}
    y_{i} = (\sum_{r_{ui} \in r_{u*}}x_{u}x_{u}^{T} +
            \lambda I_{k})^{-1}\sum_{r_{ui} \in r_{*i}}r_{ui}x_{u}
\end{equation}
\indent\indent \textbf{end for}\\
\indent\textbf{until} convergence\\
